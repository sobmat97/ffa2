const { app, BrowserWindow } = require('electron'); const path = require('path'); let win; function createWindow() { win = new BrowserWindow({ width: 800, height: 600, webPreferences: { nodeIntegration: true } }); // Sprawdź czy wprowadziłeś właściwą ścieżkę win.loadFile(path.join(__dirname, 'build', 'index.html')); // <-- To jest kluczowa linia } app.whenReady().then(() => { createWindow(); app.on('activate', () => { if (BrowserWindow.getAllWindows().length === 0) { createWindow(); } }); }); app.on('window-all-closed', () => { if (process.platform !== 'darwin') { app.quit(); } });
